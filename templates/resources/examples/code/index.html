This is the code examples page of the Tree Menu demonstration.

Here you'll find practical code examples for common use cases of the Tree Menu application.

Basic Examples

Creating a Menu Programmatically


from tree_menu.models import Menu, MenuItem

# Create a menu
main_menu = Menu.objects.create(name='main_menu')

# Create top-level items
home = MenuItem.objects.create(menu=main_menu, name='Home', url='/', is_named_url=False)
about = MenuItem.objects.create(menu=main_menu, name='About', url='/about/', is_named_url=False)
services = MenuItem.objects.create(menu=main_menu, name='Services', url='/services/', is_named_url=False)

# Create nested items
company = MenuItem.objects.create(menu=main_menu, name='Company', url='/about/company/', is_named_url=False, parent=about)
team = MenuItem.objects.create(menu=main_menu, name='Team', url='/about/team/', is_named_url=False, parent=about)

Using Named URLs


from tree_menu.models import Menu, MenuItem

# Create a menu with named URLs
main_menu = Menu.objects.create(name='main_menu')

# Create items with named URLs
home = MenuItem.objects.create(menu=main_menu, name='Home', url='home', is_named_url=True)
about = MenuItem.objects.create(menu=main_menu, name='About', url='about', is_named_url=True)
services = MenuItem.objects.create(menu=main_menu, name='Services', url='services', is_named_url=True)
